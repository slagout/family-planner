🚀 QUICK START (For Developers)
════════════════════════════════════════════════════════════════════

1. INITIALIZE DATABASE
   import { initializeDB } from './services.indexeddb';
   const db = await initializeDB({
     name: 'FamilyPlanner',
     version: 1,
     stores: { 'tasks': 'id', 'syncQueue': 'id' }
   });

2. USE OFFLINE MODE HOOK
   import { useOfflineMode } from './useOfflineMode';
   const { isOnline, hasPendingChanges, sync } = useOfflineMode();

3. USE TOUCH COMPONENTS
   import { TouchButton, TouchInput, TouchCard } from './components/*';
   <TouchButton>Click Me</TouchButton>

4. REGISTER SERVICE WORKER
   navigator.serviceWorker.register('/service-worker.js');

5. ADD MANIFEST
   <!-- In index.html -->
   <link rel="manifest" href="/manifest.json">
